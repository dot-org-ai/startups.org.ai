# Experiment Templates

Templates provide pre-configured experiments for common validation scenarios. Start with a template and customize for your concept.

```typescript
import { ExperimentTemplate } from 'startups.org.ai'

const template: ExperimentTemplate = {
  id: 'landing-page-validation',
  name: 'Landing Page Validation',
  description: 'Validate demand with a targeted landing page and waitlist',

  type: 'demand-validation',
  channel: 'landing-page',
  suggestedDuration: '2 weeks',
  suggestedBudget: 500,

  primaryMetrics: ['waitlist-signups', 'conversion-rate'],
  secondaryMetrics: ['time-on-site', 'bounce-rate'],

  successCriteriaTemplate: [
    { metric: 'conversion-rate', operator: '>=', benchmarkType: 'absolute', benchmarkValue: 2 },
    { metric: 'waitlist-signups', operator: '>=', benchmarkType: 'absolute', benchmarkValue: 100 }
  ],

  requiredAssets: ['landing-page', 'email-sequence'],

  playbook: {
    setup: [
      'Create landing page with clear value prop',
      'Set up email capture form',
      'Configure analytics tracking',
      'Set up thank-you page with next steps'
    ],
    execution: [
      'Drive traffic via chosen channel',
      'Monitor signups daily',
      'A/B test headlines if possible',
      'Collect qualitative feedback'
    ],
    analysis: [
      'Calculate conversion rate',
      'Analyze traffic sources',
      'Review time-on-page metrics',
      'Document learnings'
    ]
  }
}
```

## Using Templates

```typescript
import { createExperimentFromTemplate } from 'startups.org.ai'

const experiment = createExperimentFromTemplate({
  templateId: 'landing-page-validation',
  startupConceptId: 'cfo-ai-healthcare',
  overrides: {
    suggestedDuration: '3 weeks',
    design: {
      variants: [
        { id: 'control', name: 'Professional Tone', allocation: 50 },
        { id: 'variant-a', name: 'Casual Tone', allocation: 50 }
      ]
    }
  }
})
```
